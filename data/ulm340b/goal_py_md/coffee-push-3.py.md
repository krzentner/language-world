    def generate(self, numRows):
        """
        :type numRows: int
        :rtype: List[List[int]]
        """
        list=[]
        for i in range(0,numRows):
            innerlist=[]
            for j in range(0,i+1):
                if(j==0 or j==i):
                    innerlist.append(1)
                else:
                    innerlist.append(list[-1][j-1]+list[-1][j])
            list.append(innerlist)
        return list
