# Steps:
#  1. Put gripper above the button
#  2. Move the gripper to one side of the wall
#  3. Move the gripper towards the button until it is stopped by the wall
#  4. Move the gripper to the other side of the wall
#  5. Move the gripper towards the button until it is near the button
#  6. Lower the gripper down to the button
#  7. Close the gripper around the button
#  8. Move the gripper away from the button and over the wall
#  9. Move the gripper above the starting location
#  10. Lower the gripper down to the starting location
#  11. Open the gripper to release the button


def button_press_topdown_wall(robot):

    # Move gripper above the button
    if check("the robot's gripper is not vertically aligned with button"):
        robot.move_gripper("vertically aligned with the button")

    # Move the gripper to one side of the wall
    if check("the robot's gripper is not left of the wall"):
        robot.move_gripper("left of the wall")
    
    # Move the gripper towards the button until it is stopped by the wall
    if check("the robot's gripper is not behind the wall"):
        robot.move_gripper("near the wall")
    
    # Move the gripper to the other side of the wall
    if check("the robot's gripper is not right of the wall"):
        robot.move_gripper("right of the wall")
    
    # Move the gripper towards the button until it is near the button
    if check("the robot's gripper is not near the button"):
        robot.move_gripper("near the button")

    # Lower the gripper down to the button
    if check("the robot's gripper is not above the button"):
        robot.move_gripper("above the button")
        robot.move_gripper("near the button", close_gripper=True)

    # Move the gripper away from the button and over the wall
    if check("the robot's gripper is not above the wall"):
        robot.move_gripper("above the wall")

    # Move the gripper above the starting location
    if check("the robot's gripper is not vertically aligned with the starting location"):
        robot.move_gripper("vertically aligned with the starting location")

    # Lower the gripper down to the starting location
    if check("the robot's gripper is not above the starting location"):
        robot.move_gripper("above the starting location")
        robot.move_gripper("at the starting location", close_gripper=False)

    # Open the gripper to release the button
    if check("the robot's gripper is not open"):
        robot.move_gripper("open")
    
    print("Button pressed successfully with the wall in the way!")
